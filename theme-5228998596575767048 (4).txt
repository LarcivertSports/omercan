        // Tam ekran modunda customControls'u video ile birlikte göstermek için
        (function() {
            const video = document.getElementById('mainVideoPlayer');
            const controls = document.getElementById('customControls');
            if (!video || !controls) return;

            // controls'u video'nun hemen sonrasına taşı (aynı parentta, video ile arka arkaya)
            function moveControlsAfterVideo() {
                if (video.nextSibling !== controls) {
                    video.parentNode.insertBefore(controls, video.nextSibling);
                }
            }
            moveControlsAfterVideo();

            // Tam ekran modunda controls'u video parent'ına fixed olarak yerleştir
            function forceShowControls() {
                moveControlsAfterVideo();
                const isFullscreen =
                    document.fullscreenElement === video ||
                    document.fullscreenElement === video.parentElement ||
                    document.webkitFullscreenElement === video ||
                    document.webkitFullscreenElement === video.parentElement ||
                    document.mozFullScreenElement === video ||
                    document.mozFullScreenElement === video.parentElement ||
                    document.msFullscreenElement === video ||
                    document.msFullscreenElement === video.parentElement;
                if (isFullscreen) {
                    controls.style.display = 'flex';
                    controls.style.position = 'fixed';
                    controls.style.left = '0';
                    controls.style.right = '0';
                    controls.style.bottom = '0';
                    controls.style.zIndex = '9999';
                    controls.style.width = '100vw';
                    controls.style.pointerEvents = 'auto';
                } else {
                    controls.style.display = 'flex';
                    controls.style.position = 'absolute';
                    controls.style.left = '0';
                    controls.style.right = '0';
                    controls.style.bottom = '0';
                    controls.style.zIndex = '10';
                    controls.style.width = '100%';
                    controls.style.pointerEvents = '';
                }
            }

            // fullscreenchange event'lerinde de çalıştır
            document.addEventListener('fullscreenchange', forceShowControls);
            document.addEventListener('webkitfullscreenchange', forceShowControls);
            document.addEventListener('mozfullscreenchange', forceShowControls);
            document.addEventListener('MSFullscreenChange', forceShowControls);
            // Her frame'de kontrol et
            function rafLoop() {
                forceShowControls();
                requestAnimationFrame(rafLoop);
            }
            rafLoop();
        })();
